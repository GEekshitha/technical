scanf("%d", &n2);

for(i=n1; i<=n2; i++)
{
  sum = 0;
  temp = i;
  while(temp>0)
  {
    rem = temp%10;
    sum = sum + (rem*rem*rem);
    temp = temp/10;
  }
  if(sum == i)
    printf("%d is an Armstrong number.\n", i);
}

}
